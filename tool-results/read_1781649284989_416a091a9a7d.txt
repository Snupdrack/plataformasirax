    21→// ==================== API Helper ====================
    22→const API = {
    23→  get: async (path: string, token?: string) => {
    24→    const res = await fetch(path, { headers: token ? { Authorization: `Bearer ${token}` } : {} })
    25→    return res.json()
    26→  },
    27→  post: async (path: string, body: any, token?: string) => {
    28→    const res = await fetch(path, {
    29→      method: 'POST',
    30→      headers: { 'Content-Type': 'application/json', ...(token ? { Authorization: `Bearer ${token}` } : {}) },
    31→      body: JSON.stringify(body),
    32→    })
    33→    return res.json()
    34→  },
    35→  del: async (path: string, token: string) => {
    36→    const res = await fetch(path, { method: 'DELETE', headers: { Authorization: `Bearer ${token}` } })
    37→    return res.json()
    38→  },

    41→// ==================== Auth Context ====================
    42→type User = { id: string; email: string; full_name: string; role: string; organization?: string }
    43→
    44→const AuthContext = createContext<{
    45→  user: User | null; token: string | null; login: (t: string, u: User) => void; logout: () => void
    46→}>({ user: null, token: null, login: () => {}, logout: () => {} })
    47→
    48→function AuthProvider({ children }: { children: React.ReactNode }) {
    49→  const [user, setUser] = useState<User | null>(null)
    50→  const [token, setToken] = useState<string | null>(null)
    51→  const [ready, setReady] = useState(false)
    52→
    53→  useEffect(() => {
    54→    // Read from localStorage on mount - this is the only safe place
    55→    const t = localStorage.getItem('synkdata_token')
    56→    const u = localStorage.getItem('synkdata_user')
    57→    if (t) setToken(t)
    58→    if (u) {
    59→      try { setUser(JSON.parse(u)) } catch {}
    60→    }
    61→    setReady(true)
    62→    // eslint-disable-next-line react-hooks/set-state-in-effect
    63→  }, [])
    64→
    65→  const login = (t: string, u: User) => {
    66→    localStorage.setItem('synkdata_token', t)
    67→    localStorage.setItem('synkdata_user', JSON.stringify(u))
    68→    setToken(t); setUser(u)
    69→  }
    70→
    71→  const logout = () => {
    72→    localStorage.removeItem('synkdata_token')
    73→    localStorage.removeItem('synkdata_user')
    74→    localStorage.removeItem('synkdata_view')
    75→    setToken(null); setUser(null)
    76→  }
    77→
    78→  // Show nothing during SSR/hydration to avoid mismatch
    79→  if (!ready) {
    80→    return <div className="min-h-screen bg-white" />
    81→  }
    82→
    83→  return <AuthContext.Provider value={{ user, token, login, logout }}>{children}</AuthContext.Provider>
    84→}
    85→
    86→function useAuth() { return useContext(AuthContext) }
    87→
    88→// Helper to get token - falls back to localStorage if context token is null
    89→function useToken(): string | null {
    90→  const { token } = useAuth()
    91→  if (token) return token
    92→  if (typeof window !== 'undefined') return localStorage.getItem('synkdata_token')
    93→  return null
    94→}

  1561→// ==================== Main App ====================
  1562→export default function Home() {
  1563→  const [view, setView] = useState<View>('landing')
  1564→  const [viewData, setViewData] = useState<any>(null)
  1565→  const [mounted, setMounted] = useState(false)
  1566→  const { user, login: authLogin, logout: authLogout } = useAuth()
  1567→
  1568→  // After mount, read localStorage and set correct view
  1569→  useEffect(() => {
  1570→    const savedView = localStorage.getItem('synkdata_view') as View | null
  1571→    const token = localStorage.getItem('synkdata_token')
  1572→    if (token && user) {
  1573→      setView(savedView && savedView !== 'landing' && savedView !== 'login' && savedView !== 'register' ? savedView : 'dashboard')
  1574→    } else if (token && !user) {
  1575→      // Token exists but user not loaded yet - will be handled by auth provider
  1576→      setView('dashboard')
  1577→    } else {
  1578→      setView('landing')
  1579→    }
  1580→    setMounted(true)
  1581→    // eslint-disable-next-line react-hooks/set-state-in-effect
  1582→  }, [])
  1583→
  1584→  // Also redirect when user changes after mount
  1585→  useEffect(() => {
  1586→    if (!mounted) return
  1587→    if (user && (view === 'landing' || view === 'login' || view === 'register')) {
  1588→      setView('dashboard')
  1589→      localStorage.setItem('synkdata_view', 'dashboard')
  1590→    }
  1591→    // eslint-disable-next-line react-hooks/set-state-in-effect
  1592→  }, [user, mounted, view])
  1593→
  1594→  const navigate = useCallback((v: View, data?: any) => {
  1595→    setView(v)
  1596→    setViewData(data || null)
  1597→    if (typeof window !== 'undefined') {
  1598→      localStorage.setItem('synkdata_view', v)
  1599→    }
  1600→  }, [])
  1601→
  1602→  const renderView = () => {
  1603→    // If not mounted yet (SSR/hydration), show nothing
  1604→    if (!mounted) return <div className="min-h-screen bg-white" />
  1605→
  1606→    switch (view) {
  1607→      case 'landing': return <LandingView />
  1608→      case 'login': return user ? <DashboardLayout><DashboardView /></DashboardLayout> : <LoginView />
  1609→      case 'register': return user ? <DashboardLayout><DashboardView /></DashboardLayout> : <RegisterView />
  1610→      case 'dashboard': return <DashboardLayout><DashboardView /></DashboardLayout>
  1611→      case 'new-check': return <DashboardLayout><NewCheckView /></DashboardLayout>
  1612→      case 'check-results': return <DashboardLayout><CheckResultsView /></DashboardLayout>
  1613→      case 'history': return <DashboardLayout><HistoryView /></DashboardLayout>
  1614→      case 'curp': return <DashboardLayout><CurpValidatorView /></DashboardLayout>
  1615→      case 'rfc': return <DashboardLayout><RfcValidatorView /></DashboardLayout>
  1616→      case 'sanctions': return <DashboardLayout><SanctionsView /></DashboardLayout>
  1617→      case 'api-docs': return <DashboardLayout><ApiDocsView /></DashboardLayout>
  1618→      default: return <LandingView />
  1619→    }
  1620→  }
  1621→
  1622→  return (
  1623→    <RouterContext.Provider value={{ view, navigate, viewData }}>
  1624→      {renderView()}
  1625→    </RouterContext.Provider>
  1626→  )
  1627→}
